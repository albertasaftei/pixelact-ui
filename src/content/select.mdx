
import { Select, SelectTrigger, SelectValue, SelectContent, SelectItem } from "@/components/ui/pixelact-ui/select";

## Select

A pixel art styled select component built on top of shadcn/ui's Select. Features retro styling, pixel font, and full support for all underlying select props. Useful for dropdowns, forms, and more.

### Installation

Add the Select component to your project using the shadcn CLI:

```bash
npx shadcn@latest add @pixelact-ui/select
```

This will automatically install the component and its dependencies to your project.

### Basic Select

<CodePreview
	language="tsx"
	code={`import { Select, SelectTrigger, SelectValue, SelectContent, SelectItem } from "@/components/ui/pixelact-ui/select";

export function BasicSelect() {
	return (
		<Select>
			<SelectTrigger>
				<SelectValue placeholder="Select a fruit" />
			</SelectTrigger>
			<SelectContent>

				<SelectItem value="banana">Banana</SelectItem>

			</SelectContent>

}`}
