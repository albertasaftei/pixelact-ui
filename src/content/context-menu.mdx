## Context Menu

A pixel art styled context menu component built on top of shadcn/ui's ContextMenu. Features retro styling, pixel font, and full support for all underlying context menu props. Useful for right-click menus, actions, and more.

### Installation

Add the ContextMenu component to your project using the shadcn CLI:

```bash
npx shadcn@latest add https://pixelactui.com/r/pixelact-context-menu.json
```

This will automatically install the component and its dependencies to your project.

### Context Menu with Checkbox and Radio Items

<CodePreview
	language="tsx"
	code={`import * as React from "react";
import {
	ContextMenu,
	ContextMenuTrigger,
	ContextMenuContent,
	ContextMenuItem,
	ContextMenuCheckboxItem,
	ContextMenuRadioItem,
	ContextMenuRadioGroup,
	ContextMenuSeparator,
} from "@/components/ui/pixelact-ui/context-menu";
import { Button } from "@/components/ui/pixelact-ui/button";

export function CheckboxRadioContextMenu() {
	const [checked, setChecked] = React.useState(true);
	const [color, setColor] = React.useState("blue");
	return (
		<ContextMenu>
			<ContextMenuTrigger asChild>
				<Button>Right click for options</Button>
			</ContextMenuTrigger>
			<ContextMenuContent>
				<ContextMenuCheckboxItem checked={checked} onCheckedChange={setChecked}>
					Show Grid
				</ContextMenuCheckboxItem>
				<ContextMenuSeparator />
				<ContextMenuRadioGroup value={color} onValueChange={setColor}>
					<ContextMenuRadioItem value="blue">Blue</ContextMenuRadioItem>
					<ContextMenuRadioItem value="red">Red</ContextMenuRadioItem>
					<ContextMenuRadioItem value="green">Green</ContextMenuRadioItem>
				</ContextMenuRadioGroup>
			</ContextMenuContent>
		</ContextMenu>
	);
}`}
>
	{(() => {
		const [checked, setChecked] = React.useState(true);
		const [color, setColor] = React.useState("blue");
		return (
			<ContextMenu>
				<ContextMenuTrigger asChild>
					<Button>Right click for options</Button>
				</ContextMenuTrigger>
				<ContextMenuContent>
					<ContextMenuCheckboxItem checked={checked} onCheckedChange={setChecked}>
						Show Grid
					</ContextMenuCheckboxItem>
					<ContextMenuSeparator />
					<ContextMenuRadioGroup value={color} onValueChange={setColor}>
						<ContextMenuRadioItem value="blue">Blue</ContextMenuRadioItem>
						<ContextMenuRadioItem value="red">Red</ContextMenuRadioItem>
						<ContextMenuRadioItem value="green">Green</ContextMenuRadioItem>
					</ContextMenuRadioGroup>
